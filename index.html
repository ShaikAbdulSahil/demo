🔍 Searching Algorithms
// Linear Search O(n)
for (int i = 0; i < arr.length; i++) {
  if (arr[i] == target) return i;
}
return -1;

// Binary Search O(log n) — array must be SORTED
int lo = 0, hi = arr.length - 1;
while (lo <= hi) {
  int mid = lo + (hi - lo) / 2;
  if (arr[mid] == target) return mid;
  else if (arr[mid] < target) lo = mid + 1;
  else                          hi = mid - 1;
}

// Built-in
Arrays.binarySearch(arr, target); // returns index

      int tmp = arr[j]; arr[j]=arr[j+1]; arr[j+1]=tmp;
    }

🔧 HashSet · Stack · Queue

⚡ KEY DIFFERENCE: Array = fixed size, fast access. ArrayList = dynamic. HashMap = O(1) get/put. HashSet = no duplicates. Use Integer (wrapper) for generics, int for primitives.

📜 Interface & Abstract Class
// Interface — all methods abstract by default
public interface Flyable {
  void fly();                         // abstract
  default void land() { ... }        // default impl
}

// Abstract class — can have both
public abstract class Shape {
  abstract double area();            // must override
  public void print() { ... }        // concrete
}

// class can implement multiple interfaces

                  implements Flyable {
  public void fly() { System.out.println("Soar!"); }
}

// KEY: interface → implements, abstract → extends
// Cannot instantiate abstract class or interface

  id       INT PRIMARY KEY AUTO_INCREMENT,
  name     VARCHAR(100) NOT NULL,

🔐 DCL & TCL
-- DCL (Data Control Language)
GRANT SELECT, INSERT ON employees TO user1;
REVOKE INSERT ON employees FROM user1;

-- TCL (Transaction Control Language)
BEGIN;                     -- or START TRANSACTION
  UPDATE accounts SET bal = bal - 100 WHERE id=1;
  UPDATE accounts SET bal = bal + 100 WHERE id=2;
COMMIT;                    -- save both or none

ROLLBACK;                  -- undo if error
SAVEPOINT sp1;             -- partial rollback point
ROLLBACK TO sp1;           -- undo to savepoint

-- ACID: Atomicity, Consistency, Isolation, Durability

📊 GROUP BY, HAVING, ORDER BY
-- GROUP BY + aggregate
SELECT dept_id,
       COUNT(*) AS headcount,
       AVG(salary) AS avg_sal,
       MAX(salary), MIN(salary), SUM(salary)
FROM employees
GROUP BY dept_id
HAVING COUNT(*) > 5         -- filter AFTER grouping
ORDER BY avg_sal DESC;

-- WHERE vs HAVING
-- WHERE  filters ROWS  (before grouping)
-- HAVING filters GROUPS (after grouping)

SELECT dept_id, COUNT(*)
FROM employees
WHERE salary > 30000     -- rows filter
GROUP BY dept_id
HAVING COUNT(*) >= 3;   -- group filter

🔗 JOINs — Most Asked!
-- INNER JOIN — only matching rows both tables
SELECT e.name, d.dept_name
FROM employees e
INNER JOIN departments d ON e.dept_id = d.id;

-- LEFT JOIN — all from left, matching from right (NULL if no match)
SELECT e.name, d.dept_name
FROM employees e
LEFT JOIN departments d ON e.dept_id = d.id;

-- RIGHT JOIN — all from right, matching from left
SELECT e.name, d.dept_name
FROM employees e
RIGHT JOIN departments d ON e.dept_id = d.id;

-- FULL OUTER JOIN — all rows from both
SELECT * FROM A FULL OUTER JOIN B ON A.id = B.id;

-- SELF JOIN — join table with itself
SELECT e.name, m.name AS manager
FROM employees e
JOIN employees m ON e.manager_id = m.id;

-- CROSS JOIN — cartesian product
SELECT * FROM A CROSS JOIN B;
🔭 Subqueries

SELECT dept_id, AVG(sal)

⚡ Quick Reference: SELECT Order
-- Always write in this order:
SELECT   columns / agg functions
FROM     table
JOIN     other_table ON condition
WHERE    row_filter        ← before grouping
GROUP BY columns
HAVING   group_filter      ← after grouping
ORDER BY column ASC/DESC
LIMIT    n;

-- Execution order (how DB runs it):
-- FROM → JOIN → WHERE → GROUP BY
--   → HAVING → SELECT → ORDER BY → LIMIT

-- DISTINCT
